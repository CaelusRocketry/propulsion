14/* Function: rt_BackwardSubstitutionRR_Dbl =====================================



15 * Abstract: Backward substitution: Solving Ux=b 



16 *           U: real, double



17 *           b: real, double



18 *           U is an upper (or unit upper) triangular full matrix.



19 *           The entries in the lower triangle are ignored.



20 *           U is a NxN matrix



21 *           X is a NxP matrix



22 *           B is a NxP matrix



23 */



24void rt_BackwardSubstitutionRR_Dbl(real_T          *pU,



25                                   const real_T    *pb,



26                                   real_T          *x,



27                                   int_T            N,



28                                   int_T            P,



29                                   boolean_T        unit_upper)



30{



31  int_T i,k;



32  for(k=P; k>0; k--) {



33    real_T *pUcol = pU;



34    for(i=0; i<N; i++) {



35      real_T *xj = x + k*N-1;



36      real_T s = 0.0;



37      real_T *pUrow = pUcol--;          /* access current row of U */



38



39      {



40        int_T j = i;



41        while(j-- > 0) {



42          s += *pUrow * *xj--;



43          pUrow -= N;



44        }



45      }



46



47      if (unit_upper) {



48        *xj = *pb-- - s;



49      } else {



50        *xj = (*pb-- - s) / *pUrow;



51      }



52    }



53  }



54}
